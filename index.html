Space Explorer Game

This is a simple game where you dodge falling obstacles. Use the left/right arrow keys to move.

Score: 0

How This Game Was Made:

This game was created using HTML, CSS, and JavaScript. Here's the code:


<!-- HTML Structure -->
<div id="game-container">
  <div id="player"></div>
  <div class="score-display">Score: <span id="score">0</span></div>
</div>

<!-- CSS Styling -->
<style>
  #game-container {
    width: 100%;
    height: 400px;
    background: #333;
    position: relative;
    overflow: hidden;
  }

  #player {
    width: 50px;
    height: 50px;
    background-color: #f72585;
    border-radius: 50%;
    position: absolute;
    bottom: 20px;
  }

  .obstacle {
    width: 60px;
    height: 20px;
    background-color: #7209b7;
    position: absolute;
  }

  .score-display {
    position: absolute;
    top: 10px;
    right: 10px;
    background: rgba(255,255,255,0.8);
    padding: 5px 10px;
    border-radius: 5px;
  }
</style>

<!-- JavaScript Game Logic -->
<script>
  // Game variables
  let score = 0;
  let playerX = 175;
  let gameActive = true;
  
  // Get elements
  const player = document.getElementById('player');
  const gameContainer = document.getElementById('game-container');
  const scoreDisplay = document.getElementById('score');
  
  // Set initial player position
  player.style.left = playerX + 'px';
  
  // Handle keyboard controls
  document.addEventListener('keydown', (e) => {
    if (!gameActive) return;
    
    if (e.key === 'ArrowLeft' && playerX > 0) {
      playerX -= 15;
    } else if (e.key === 'ArrowRight' && playerX < gameContainer.clientWidth - 50) {
      playerX += 15;
    }
    
    player.style.left = playerX + 'px';
  });
  
  // Create obstacles
  function createObstacle() {
    if (!gameActive) return;
    
    const obstacle = document.createElement('div');
    obstacle.classList.add('obstacle');
    
    const randomX = Math.floor(Math.random() * (gameContainer.clientWidth - 60));
    obstacle.style.left = randomX + 'px';
    
    gameContainer.appendChild(obstacle);
    
    let obstacleY = -20;
    
    // Animate obstacle falling
    const obstacleInterval = setInterval(() => {
      if (!gameActive) {
        clearInterval(obstacleInterval);
        return;
      }
      
      obstacleY += 5;
      obstacle.style.top = obstacleY + 'px';
      
      // Check for collision
      if (
        obstacleY + 20 >= gameContainer.clientHeight - 70 &&
        obstacleY <= gameContainer.clientHeight - 20 &&
        randomX + 60 >= playerX &&
        randomX <= playerX + 50
      ) {
        gameOver();
        clearInterval(obstacleInterval);
      }
      
      // Check if obstacle passed player
      if (obstacleY > gameContainer.clientHeight) {
        score++;
        scoreDisplay.textContent = score;
        obstacle.remove();
        clearInterval(obstacleInterval);
      }
    }, 50);
  }
  
  // Game over function
  function gameOver() {
    gameActive = false;
    alert('Game Over! Your score: ' + score);
    restartGame();
  }
  
  // Restart game
  function restartGame() {
    score = 0;
    scoreDisplay.textContent = score;
    playerX = 175;
    player.style.left = playerX + 'px';
    
    // Remove all obstacles
    document.querySelectorAll('.obstacle').forEach(obs => obs.remove());
    
    gameActive = true;
  }
  
  // Start creating obstacles
  setInterval(createObstacle, 1500);
</script>
